## File 类的应用

```java
package com.junjun.file;
import java.io.File;
import java.io.IOException;
public class FileDemo {
  public static void main(String[] args) {
    // 绝对路径
    // File file = new File("/Users/lijunjun/MKW/Java-2021/13-输入输出流/src/com/junjun/file/FileDemo.java");
    File file1 = new File("/Users/lijunjun/MKW/Java-2021/13-输入输出流/src/com/junjun/file", "/index.txt");
    System.out.println("是否是文件: " + file1.isFile());
    System.out.println("是否是目录: " + file1.isDirectory());

    // 创建文件
    File file2 = new File("/Users/lijunjun/MKW/Java-2021/13-输入输出流/src/com/junjun/file", "/set/HashSet");
    if (!file2. ()) {
      // 创建一级目录
      // file2.mkdir();

      // 如果是多级目录,要加s
      file2.mkdirs();
    }

    if (!file1.exists()) {
      // 创建这个文件
      try {
        file1.createNewFile();
      } catch (IOException e) {
        e.printStackTrace();
      }
    }
  }
}
```

```java
public class FileInputDemo1 {
  public static void main(String[] args) {
    try {
      FileInputStream fs = new FileInputStream("index.txt");
      // int n = fs.read();
      // while (n != -1) {
      //   System.out.print((char) n);
      //   n = fs.read();
      // }
      // 简写,输出结果
      int n = 0;
      while ((n = fs.read()) != -1) {
        System.out.print((char) n);
      }
      // System.out.println((char) n);
      // 关闭资源 
      fs.close();
    } catch (IOException e) {
      e.printStackTrace();
    }
  }
}
```

- demo2

```java
public class FileInputDemo2 {
  public static void main(String[] args) {
    try {
      FileInputStream fs = new FileInputStream("index.txt");
      byte[] b = new byte[100];
      fs.read(b,0,5);
      System.out.println(new String(b));
      fs.close();
    } catch (IOException e) {
      e.printStackTrace();
    }
  }
}
```

## FileOutputStream

```java
package com.junjun.output;
import java.io.FileOutputStream;
import java.io.IOException;
public class FileOutPutDemo1 {
  public static void main(String[] args) {
    try {
      FileOutputStream fo = new FileOutputStream("index.txt");
      fo.write(50);
      fo.write('a');
      fo.close();
    } catch (IOException e) {
      e.printStackTrace();
    }
  }
}
```
